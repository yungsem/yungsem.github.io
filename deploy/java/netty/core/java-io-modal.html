# Java 网络 IO 模型

# BIO

工作模型：服务端为每一个客户端单独开启一个线程。

示意图表示如下：

代码示例：

package com.yungsem.springbootnetty.bio;

import lombok.extern.slf4j.Slf4j;

import java.io.IOException;
import java.io.InputStream;
import java.net.ServerSocket;
import java.net.Socket;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;

/**
 * @author yangsen
 * @version 1.0
 * @since 2020-08-07
 */
@Slf4j
public class Server {

    public static void main(String[] args) throws IOException {

        ExecutorService cachedThreadPool = Executors.newCachedThreadPool();

        // 创建 serverSocket
        // 绑定端口 6666
        ServerSocket serverSocket = new ServerSocket(6666);

        while (true) {
            // 监听 6666 端口，等待客户端连接
            log.info("等待连接。。。");
            Socket socket = serverSocket.accept();
            log.info("连接成功。。。");

            // 如果有新的客户端成功连接进来，就单独开启一个线程处理和这个客户端的通信
            cachedThreadPool.execute(() -> {
                printStringFromSocket(socket);
            });
        }
    }

    /**
     * 从 socket 中读取消息并打印
     *
     * @param socket socket
     */
    private static void printStringFromSocket(Socket socket) {
        InputStream ins = null;
        try {
            log.info("线程{}等待读数据。。。", Thread.currentThread().getName());
            ins = socket.getInputStream();
            byte[] bytes = new byte[1024];
            while (true) {
                int read = ins.read(bytes);
                if (read != -1) {
                    log.info("线程{}读数据成功，数据：{}", Thread.currentThread().getName(), new String(bytes));
                } else {
                    break;
                }
            }
        } catch (IOException e) {
            log.error("读取消息异常", e);
        } finally {
            // 关闭 socket
            try {
                socket.close();
            } catch (IOException e) {
                log.error("关闭 socket 异常", e);
            }
            // 关闭输入流
            if (ins != null) {
                try {
                    ins.close();
                } catch (IOException e) {
                    log.error("关闭 socket 输入流异常", e);
                }
            }
        }
    }
}

# NIO

工作模型：服务端维护一个线程，使用一个 Selector ，将多个客户端对应的 SocketChannel 都注册到这个 Selector 上，Seletor 不断的轮询，如果某个 SocketChannel 可读或者可写了，就处理，否则一直轮询。

示意图表示如下：

代码示例：

package com.yungsem.springbootnetty.nio;

import lombok.extern.slf4j.Slf4j;

import java.io.IOException;
import java.net.InetSocketAddress;
import java.nio.ByteBuffer;
import java.nio.channels.*;
import java.util.Set;

/**
 * @author yangsen
 * @version 1.0
 * @since 2020-08-08
 */
@Slf4j
public class Server {

    public static void main(String[] args) throws IOException {
        // 创建 selector
        Selector selector = Selector.open();

        // 创建 serverSocketChannel
        ServerSocketChannel serverSocketChannel = ServerSocketChannel.open();
        // 绑定端口
        serverSocketChannel.socket().bind(new InetSocketAddress(6666));
        // 设置为非阻塞
        serverSocketChannel.configureBlocking(false);

        // 将 serverSocketChannel 注册到 selector 上，关心事件为 OP_ACCEPT
        serverSocketChannel.register(selector, SelectionKey.OP_ACCEPT);

        while (true) {
            int select = selector.select(1000);
            if (select == 0) { // 没有事件发生
                log.info("等待了 1 秒，没有事件发生");
                continue;
            }

            // 有事件发生
            // 获取已经发生的事件
            Set<SelectionKey> selectionKeys = selector.selectedKeys();
            for (SelectionKey selectionKey : selectionKeys) {
                if (selectionKey.isAcceptable()) { // OP_ACCEPT 事件，有新连接进来
                    log.info("有新连接进来");
                    // 建立连接，获取 socketChannel
                    SocketChannel socketChannel = serverSocketChannel.accept();
                    socketChannel.configureBlocking(false);
                    // 将 socketChannel 注册到 selector 上，关心事件为 OP_READ ，同时为该 socketChannel 绑定一个 buffer
                    socketChannel.register(selector, SelectionKey.OP_READ, ByteBuffer.allocate(1024));
                }
                if (selectionKey.isReadable()) { // OP_READ 事件，客户端发来的消息已经可读
                    // 获取 socketChannel
                    SocketChannel socketChannel = (SocketChannel)selectionKey.channel();
                    // 获取 socketChannel 的 buffer
                    ByteBuffer buffer = (ByteBuffer)selectionKey.attachment();
                    // 将 socketChannel 中的数据读取到 buffer 中
                    socketChannel.read(buffer);
                    log.info("读取到消息：{}", new String(buffer.array()));
                }

                // selectionKey 处理完之后，要将 selectionKey 从 selectionKeys 中删除
                // 因为本次就绪的事件都已经处理完毕，如果不删，会造成重复处理同一事件
                // 新事件就绪时，下一次 while 循环会自动捕获到
                selectionKeys.remove(selectionKey);
            }
        }
    }
}

package com.yungsem.springbootnetty.nio;

import lombok.extern.slf4j.Slf4j;

import java.io.IOException;
import java.net.InetSocketAddress;
import java.nio.ByteBuffer;
import java.nio.channels.SocketChannel;

/**
 * @author yangsen
 * @version 1.0
 * @since 2020-08-08
 */
@Slf4j
public class Client {

    public static void main(String[] args) throws IOException {
        // 创建 socketChannel
        SocketChannel socketChannel = SocketChannel.open();
        // 设置为非阻塞模式
        socketChannel.configureBlocking(false);

        // 连接服务端
        boolean connect = socketChannel.connect(new InetSocketAddress("127.0.0.1", 6666));
        // 如果没有连接成功，程序不会阻塞
        // 循环打印日志，等待连接成功
        if (!connect) {
            while (!socketChannel.finishConnect()) {
                log.info("等待连接建立...");
            }
        }

        // 连接成功，发送数据
        String message = "Hello, Server!";
        ByteBuffer buffer = ByteBuffer.wrap(message.getBytes());
        socketChannel.write(buffer);

        // 为了让程序等待
        System.in.read();
    }
}
