# GMP初始化

# 程序入口

// src/runtime/rt0_linux_amd64.s

TEXT _rt0_amd64_linux(SB),NOSPLIT,$-8
	JMP	_rt0_amd64(SB)

跳转到函数 _rt0_amd64(SB) 。

# 处理 argc 和 argv

// src/runtime/asm_amd64.s

TEXT _rt0_amd64(SB),NOSPLIT,$-8
	MOVQ	0(SP), DI
	LEAQ	8(SP), SI
	JMP	runtime·rt0_go(SB)

主线程开始执行时，会将两个参数压入主线程栈的栈顶，两个参数分别是 argc 和 argv 。argc 是 int 类型，在 64 位系统里占 8 个字节，位于主线程栈的栈顶，即 0(SP) 位置。argv 是一个数组，占用多少个字节不确定，它位于 argc 的上面，所以起始位置是 8(SP) 。

L1：将 argc 的值存入 DI 中。

L2：将 arv 的地址，也即 8(SP) 处的地址存入 SI 中。

然后跳转到 runtime·rt0_go(SB) 函数。

# src/runtime/asm_amd64.s

函数 runtime·rt0_go(SB) 位于文件 src/runtime/asm_amd64.s 中，下面是对该函数的源码分析，比较长，分段进行。

# 调整 argc 和 argv 的位置，SP 内存对齐

MOVQ	DI, AX
MOVQ	SI, BX
SUBQ	$(4*8+7), SP
ANDQ	$~15, SP
MOVQ	AX, 16(SP)
MOVQ	BX, 24(SP)

L1：argc 的值在上面被存在 DI 中，现在从 DI 中转移到 AX 中。

L2：argv[] 的起始地址在上面被存在 SI 中，现在从 SI 中转移到 BX 中。

L3~4：内存对齐，将 SP 移动到 16 字节倍数的位置。

L5：AX 中是 argc 的值，现在移动到 16(SP) 位置。

L6：BX 中是 argv[] 的起始地址，现在移动到 24(SP) 位置。这样一来，SP 与 argc 中间空了 16 字节。

# 初始化 g0 栈

MOVQ	$runtime·g0(SB), DI
LEAQ	(-64*1024+104)(SP), BX
MOVQ	BX, g_stackguard0(DI)
MOVQ	BX, g_stackguard1(DI)
MOVQ	BX, (g_stack+stack_lo)(DI)
MOVQ	SP, (g_stack+stack_hi)(DI)

L1：将全局变量 g0 的地址放入 DI 中。

L2：BX = SP - 64*1024 + 104 。

L3：g_stackguard0(DI) 就表示 g0.stackguard0 ，该句作用是让 g0.stackguard0 = BX 。

L4： g0.stackguard1 = BX

L5： g0.stack.lo = BX

L6： g0.stack.hi = SP

# 初始化线程本地存储

LEAQ	runtime·m0+m_tls(SB), DI
CALL	runtime·settls(SB)

// store through it, to make sure it works
get_tls(BX)
MOVQ	$0x123, g(BX)
MOVQ	runtime·m0+m_tls(SB), AX
CMPQ	AX, $0x123
JEQ 2(PC)
CALL	runtime·abort(SB)

L1：将 m0.tls 的地址放入 DI 中。

L2：调用 runtime·settls(SB) ，该函数位于文件 src/runtime/sys_linux_amd64.s 中，分析见下面。

剩下的代码用于检验线程本地存储是否起作用了。

# runtime·settls(SB)

TEXT runtime·settls(SB),NOSPLIT,$32
#ifdef GOOS_android
	SUBQ	runtime·tls_g(SB), DI
#else
  // DI 原本存放的是 m0.tls 的地址，也即 m0.tls[0] 的地址
  // 现在 DI = DI + 8 ，也即 DI 存放 m0.tls[1] 的地址
  ADDQ	$8, DI	// ELF wants to use -8(FS)
#endif
  // arch_prctl 系统调用的第二个参数
  MOVQ	DI, SI
  // arch_prctl 系统调用的第一个参数
  MOVQ	$0x1002, DI	// ARCH_SET_FS
  // arch_prctl 系统调用的调用编号
  MOVQ	$SYS_arch_prctl, AX
  // 系统调用
  SYSCALL
  CMPQ	AX, $0xfffffffffffff001
  JLS	2(PC)
  MOVL	$0xf1, 0xf1  // crash
  RET

系统 rch_prctl 将 FS 段寄存器的值设置为 m0.tls[1] 的地址。即 FS = &m0.tls[1] 。 每个线程的 FS 段寄存器值是不一样的，所以每个线程都有自己的 m0.tls[1] 地址，即线程本地存储。

# 关联 m0 和 g0

// BX = &m0.tls[0]
get_tls(BX)
// CX = &g0
LEAQ    runtime·g0(SB), CX
// m0.tls[0] = &go
MOVQ    CX, g(BX)
// AX = &m0
LEAQ    runtime·m0(SB), AX

// save m->g0 = g0
// m.g0 = &g0
MOVQ    CX, m_g0(AX)
// save m0 to g0->m
// g0.m = &m0
MOVQ    AX, g_m(CX)

CLD // convention is D is always left cleared
CALL    runtime·check(SB)

结果：

m0.tls[0] = &g0
m.g0 = &g0
g0.m = &m0

# 初始化调度器

MOVL	16(SP), AX		// copy argc
MOVL	AX, 0(SP)
MOVQ	24(SP), AX		// copy argv
MOVQ	AX, 8(SP)
CALL	runtime·args(SB)
CALL	runtime·osinit(SB)
CALL	runtime·schedinit(SB)

L1：将 argc 转移到 AX 中

L2：将 AX 中的内容，也即 argc 转移到 0(SP) 中

L3：将 argv 数组的首地址转移到 AX 中

L4：将 AX 中的内容，也即 argv 数组的首地址转移到 8(SP) 中

L5：调用 runtime·args(SB)

L6：调用 runtime·osinit(SB)

L7：调用 runtime·schedinit(SB)

# 调用 runtime·schedinit(SB)

该函数位于 src/runtime/proc.go 文件中。其主要内容分析如下：

// 获取当前的 g ，初始化时就是 g0
_g_ := getg()

// 设置全局变量 sched 的 maxmcount = 10000
// 也即最多 10000 个 m
sched.maxmcount = 10000

// 调用 mcommoninit ，初始化 m0
mcommoninit(_g_.m, -1)

// 设置 procs ，等于 CPU 核心数
procs := ncpu

if n, ok := atoi32(gogetenv("GOMAXPROCS")); ok && n > 0 {
procs = n
}

// 调用 procresize
if procresize(procs) != nil {
throw("unknown runnable goroutine during bootstrap")
}

# 调用 mcommoninit

// 获取当前的 g ，初始化时就是 g0
_g_ := getg()

// 初始化时，mp 就是 m0
// 初始化时，m0.id = 0
if id >= 0 {
mp.id = id
} else {
// 简单来说，就是 id 递增，然后要检查 id 是否超过 maxmcount 
mp.id = mReserveID()
}

// 初始化时，将 m0.alllink = allm
mp.alllink = allm

# 调用 procresize

使用 make([]p, nprocs) 初始化全局变量 allp，即 allp = make([]p, nprocs) 。

循环创建并初始化 nprocs 个 p 结构体对象并依次保存在 allp 切片之中。

把 m0 和 allp[0] 绑定在一起，即 m0.p = allp[0]，allp[0].m = m0 。

把除了 allp[0] 之外的所有 p 放入到全局变量 sched 的 pidle 空闲队列之中。
